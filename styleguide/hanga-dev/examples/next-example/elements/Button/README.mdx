# Button

This is a button that can be used for interactivity or linking.

## Examples

<Presets
  defaultProps={{
    href: '',
    fullWidth: false,
    size: 'default',
    theme: 'passive',

>
  <Presets.Option
    name="Basic Button"
    props={{
      children: 'This is a basic button'
    }}
  />
  <Presets.Option
    name="Large Positive Button"
    props={{
      size: 'large',
      theme: 'positive',
      children: 'This is a large positive button'
    }}
  />
  <Presets.Option
    name="Small Negative Button"
    props={{
      size: 'small',
      theme: 'negative',
      children: 'This is a small negative button'
    }}
  />
  <Presets.Option
    name="Full Width Button Link"
    props={{
      href: '#wow',
      fullWidth: true,
      children: 'This is a wide button that acts as a link'
    }}
  />
</Presets>
